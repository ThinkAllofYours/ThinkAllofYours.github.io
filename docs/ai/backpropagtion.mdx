from data_prep import features, targets, features_test, targets_test

# ì¬í˜„ì„±ì„ ìœ„í•œ ëœë¤ ì‹œë“œ ì„¤ì •
np.random.seed(21)

    """

    del_w_input_hidden = np.zeros(weights_input_hidden.shape)

    for x, y in zip(features.values, targets):
        ### ìˆœì „íŒŒ(Forward Pass) ###
        # ì…ë ¥ì¸µ -> ì€ë‹‰ì¸µ
        hidden_input = np.dot(x, weights_input_hidden)    # ì€ë‹‰ì¸µ ì…ë ¥ ê³„ì‚°

        # ì€ë‹‰ì¸µ -> ì¶œë ¥ì¸µ
        output = sigmoid(np.dot(hidden_output, weights_hidden_output))  # ìµœì¢… ì¶œë ¥
        
        ### ì—­ì „íŒŒ(Backward Pass) ###
        # ì¶œë ¥ì¸µ ì˜¤ì°¨ ê³„ì‚°

        # ì¶œë ¥ì¸µì˜ ë¸íƒ€ ê³„ì‚°
        # Î´ = error * output * (1 - output)
        output_error_term = error * output * (1 - output)
        
        # ì€ë‹‰ì¸µìœ¼ë¡œ ì˜¤ì°¨ ì „íŒŒ
        # ì€ë‹‰ì¸µì´ ì¶œë ¥ì¸µ ì˜¤ì°¨ì— ê¸°ì—¬í•œ ì •ë„ ê³„ì‚°
        hidden_error = np.dot(output_error_term, weights_hidden_output)
        
        # ì€ë‹‰ì¸µì˜ ë¸íƒ€ ê³„ì‚°
        # Î´ = hidden_error * hidden_output * (1 - hidden_output)
        hidden_error_term = hidden_error * hidden_output * (1 - hidden_output)
        
        # ê°€ì¤‘ì¹˜ ë³€í™”ëŸ‰ ëˆ„ì 
        del_w_hidden_output += output_error_term * hidden_output  # ì€ë‹‰ì¸µ->ì¶œë ¥ì¸µ ê°€ì¤‘ì¹˜
        del_w_input_hidden += hidden_error_term * x[:, None]     # ì…ë ¥ì¸µ->ì€ë‹‰ì¸µ ê°€ì¤‘ì¹˜
    
    # ë¯¸ë‹ˆë°°ì¹˜ì˜ í‰ê· ìœ¼ë¡œ ê°€ì¤‘ì¹˜ ì—…ë°ì´íŠ¸
    weights_input_hidden += learnrate * del_w_input_hidden / n_records
    weights_hidden_output += learnrate * del_w_hidden_output / n_records
    
    # í›ˆë ¨ ê³¼ì • ëª¨ë‹ˆí„°ë§ - ë§¤ 10%ë§ˆë‹¤ ì†ì‹¤ ì¶œë ¥
    if e % (epochs / 10) == 0:

        out = sigmoid(np.dot(hidden_output, weights_hidden_output))
        loss = np.mean((out - targets) ** 2)  # MSE ì†ì‹¤ ê³„ì‚°
        
        # ì†ì‹¤ì´ ì¦ê°€í•˜ëŠ”ì§€ í™•ì¸ (ê²½ê³  ì¶œë ¥)
        if last_loss and last_loss < loss:

        else:
            print("Train loss: ", loss)
        last_loss = loss

# ìµœì¢… í…ŒìŠ¤íŠ¸ ì„¸íŠ¸ ì„±ëŠ¥ í‰ê°€
hidden = sigmoid(np.dot(features_test, weights_input_hidden))
