    weights_input_hidden += learnrate * del_w_input_hidden / n_records

    if e % (epochs / 10) == 0:

# ìµœì¢… í…ŒìŠ¤íŠ¸ ì„¸íŠ¸ ì„±ëŠ¥ í‰ê°€
