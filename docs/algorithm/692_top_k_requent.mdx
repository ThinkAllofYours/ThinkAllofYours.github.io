```python
class Solution:
    def topKFrequent(self, words: List[str], k: int) -> List[str]:
        dic = {}
        for word in words:
            if word in dic:
                dic[word] -= 1
            else:
                dic[word] = -1

        heap = []
        for key, value in dic.items():
            heapq.heappush(heap, (value, key))

        res = []
        for i in range(0, k):
            item = heapq.heappop(heap)
            res.append(item[1])

        return res
```
